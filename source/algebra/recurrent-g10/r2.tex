\section*{Рекуррентные соотношения}

% $authors:
% - Владимир Трушков

\begin{problems}

\item
Последовательность $\{ a_n \}$ определена как
\[
    a_1 = a_2 = 1
,\quad
    a_{n+2} = a_{n+1} + \frac{a_n}{3^n}
        ,\; n \geq 1
\, . \]
Докажите, что $a_n < 2$ при $n \geq 1$.

\item
Последовательность $\{ a_n \}$ задана рекуррентно как
\[
    a_1 = 2
,\quad
    a_n = \frac{n+1}{n-1} \cdot (a_1 + a_2 + \ldots + a_{n-1})
    ,\;
    n \geq 2
\, . \]
Найдите $a_{2013}$.

\item
Пусть $\{ a_n \}$~— последовательность, заданная как
\[
    a_1 = 2013
,\quad
    a_{n+1} = \frac{a_n}{1 + n \cdot a_n}
        ,\; n \geq 1
\, . \]
Найдите $a_{2013}$.

\item
Найдите $a_n$, если
\[
    a_0 = \frac{1}{2}
,\quad
    a_{n+1} = \frac{a_n}{2 + 3^n \cdot a_n}
        ,\; n \geq 1
\, . \]

\item
Последовательность $\{ a_n \}$ определена как
\[
    a_1 = \frac{1}{2}
,\quad
    a_m = \frac{a_{m-1}}{2 m \cdot a_{m-1} + 1}
        ,\; m > 1
.\]
Найдите значение выражения $a_1 + a_2 + \ldots + a_k$.

\item
Докажите, что если
\[
    u_0 = 0{,}001
,\quad
    u_{n+1} = u_n \cdot (1 - u_n)
        ,\; n \geq 0
\, , \]
то~$u_{1000} < 1 / 2000$.

\item
Последовательность $\{ a_n \}$ определена как
\[
    a_1 = a_2 = 1
,\quad
    a_{n+2} = a_{n+1} + \frac{a_n}{3^n}
        ,\; n \geq 1
\, . \]
Докажите, что $a_n < 2$ при $n \geq 1$.

\item
Последовательность $a_n$ определена следующим образом:
\[
    a_0 = 1 ,\; a_1 = 1
,\quad
    a_{n+1} = \frac{1 + a_{n}^2}{a_{n-1}}
\, . \]
Докажите, что все члены этой последовательности~— целые числа.

\item
Докажите, что если
\[
    a_1 = 1
,\quad
    a_{n+1} = \frac{a_n}{2} + \frac{1}{a_n}
        ,\; n \geq 1
\, , \]
то $0 < a_{10} - \sqrt{2} < 10^{-210}$.

\item
Последовательность задана условиями
\[
    a_1 = 100
,\quad
    a_{n+1} = a_n + \frac{1}{a_n}
        ,\; n \geq 1
.\]
Найдите $[a_{2015}]$.

\end{problems}
