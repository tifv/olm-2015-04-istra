\section*{Рекуррентные соотношения --- 2}

% $authors:
% - Владимир Трушков

\begin{problems}

\item
Последовательность $\{ a_n \}$ задана рекуррентно как
\[
    a_1 = 2
,\quad
    a_n = \frac{n+1}{n-1} \cdot (a_1 + a_2 + \ldots + a_{n-1})
    ,\;
    n \geq 2
\, . \]
Найдите $a_{2013}$.

\item
Найдите $a_n$, если
\[
    a_0 = \frac{1}{2}
,\quad
    a_{n+1} = \frac{a_n}{2 + 3^n \cdot a_n}
        ,\; n \geq 1
\, . \]

\item
Пусть $\{ a_n \}$~--- последовательность, заданная как
\[
    a_1 = 2013
,\quad
    a_{n+1} = \frac{a_n}{1 + n^3 \cdot a_n}
        ,\; n \geq 1
\, . \]
Найдите $a_{2013}$.

\item
Последовательность $\{ a_n \}$ определена как
\[
    a_1 = \frac{1}{2}
,\quad
    a_m = \frac{a_{m-1}}{2 m \cdot a_{m-1} + 1}
        ,\; m > 1
.\]
Найдите значение выражения $a_1 + a_2 + \ldots + a_k$.

\item
Последовательность $a_n$ определена следующим образом:
\[
    a_0 = 1 ,\; a_1 = 1
,\quad
    a_{n+1} = \frac{1 + a_{n}^2}{a_{n-1}}
\, . \]
Докажите, что все члены этой последовательности~--- целые числа.

\item
Последовательность $\{ u_n \}$ определена как
\[
    u_1 > \frac{1}{2} ,\; u_2 = 2 u_1
,\quad
    u_{n+2} = \frac{u_{n+1} + 2 u_{n+1}^2}{2 u_{n}}
\, . \]
Докажите, что
$\frac{1}{u_1} + \frac{1}{u_2} + \ldots + \frac{1}{u_n} < 4$.

\item
Последовательность $\{ a_n \}$ определена как
\[
    a_1 = 1 ,\; a_2 = 1 ,\; a_3 = 1
,\quad
    a_{n+1} = \frac{1 + a_n \cdot a_{n-1}}{a_{n-2}}
        ,\; n \geq 3
\, . \]
Докажите, что все члены последовательности~--- целые числа.

\item
Пусть $\{ x_n \}$, $n \geq 1$,~--- последовательность, заданная как
\[
    x_1 = 1
,\quad
    2 x_{n+1} = 3 x_n + \sqrt{5 x_n^{2} - 4}
\, . \]
Докажите, что последовательность состоит только из натуральных чисел.

\item
Последовательность $\{ x_n \}$ задана как
\[
    x_1 = x_2 = x_3 = 1
,\quad
    x_n = \frac{x_{n-1}^2 + x_{n-1} \cdot x_{n-2} + x_{n-2}^2}{x_{n-3}}
        ,\; n \geq 4
.\]
Докажите, что все члены этой последовательности~--- целые числа.

\item
Последовательность $\{a_n\}$ определена как
\[
    a_1 = 1
,\quad
    a_{n+1} = \left( 1 + \frac{k}{n} \right) \cdot a_n + 1
\, . \]
Найдите все натуральные~$k$, при которых все члены последовательности $a_n$~---
целые.

\end{problems}
